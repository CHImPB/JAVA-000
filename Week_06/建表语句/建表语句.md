- 用户表

```mysql
CREATE TABLE `tb_user` (
  `user_id` varchar(40) NOT NULL,
  `user_name` varchar(40) DEFAULT NULL,
  `user_password` varchar(20) DEFAULT NULL,
  `nickname` varchar(40) DEFAULT NULL,
  `IDCard` varchar(40) DEFAULT NULL,
  `Creator_Id` varchar(40) DEFAULT NULL,
  `Creator_Name` varchar(40) DEFAULT NULL,
  `Create_time` datetime DEFAULT NULL,
  PRIMARY KEY (`user_id`)
)
```

- 订单表

```mysql
 CREATE TABLE `tb_order` (
  `ORDER_Id` varchar(40) NOT NULL,
  `ORDER_CODE` varchar(40) DEFAULT NULL,
  `ORDER_TIME` datetime DEFAULT NULL,
  `ADDRESS` varchar(400) DEFAULT NULL,
  `total_amount` decimal(19,4) DEFAULT NULL,
  `user_id` varchar(40) DEFAULT NULL,
  PRIMARY KEY (`ORDER_Id`)
)
```

- 订单明细表

```mysql
CREATE TABLE `tb_order_detail` (
  `order_detail_id` varchar(40) NOT NULL,
  `order_id` varchar(40) DEFAULT NULL,
  `product_id` varchar(40) DEFAULT NULL,
  `price` decimal(19,4) DEFAULT NULL,
  `num` int(11) DEFAULT NULL,
  `discount` decimal(3,2) DEFAULT NULL,
  `total_amount` decimal(19,4) DEFAULT NULL,
  PRIMARY KEY (`order_detail_id`)
)
```

- 商品表

```mysql
CREATE TABLE `tb_product` (
  `PRODUCT_Id` varchar(40) NOT NULL,
  `CODE` varchar(40) DEFAULT NULL,
  `NAME` varchar(40) DEFAULT NULL,
  `SPEC` varchar(400) DEFAULT NULL,
  `PRICE` decimal(19,4) DEFAULT NULL,
  PRIMARY KEY (`PRODUCT_Id`)
)
```
